    int arr[100], b[100];

    printf("Enter elements in array: ");
    for(i=0; i<size; i++)
    {
        scanf("%d", &arr[i]);

        
        b[i] = -1;
    }
    occurance(size,arr,b);

    return 0;
}
int occurance(int size,int arr[],int b[])
{
    int i,j,count;
        for(i=0; i<size; i++)
    {
        count = 1;
        for(j=i+1; j<size; j++)
        {
            
            if(arr[i]==arr[j])
            {
                count++;       
                b[j] = 0;
            }
        }   
        if(b[i] != 0)
        {
            b[i] = count;
        }
    }
    printf("\noccurance of all elements of array : \n");
    for(i=0; i<size; i++)
    {
        if(b[i] != 0)
        {
            printf("%d occur %d times \n", arr[i], b[i]);
        }
    }
}

int triangle(int n)
{
    int i,j;
    for(i=1; i<=n; i++)
       {   
         for(j=i; j<n; j++)
         {
            printf(" ");
         }
         for(j=1; j<=(2*i-1); j++)
         {
            printf("*");
         }
         printf("\n");
    
       }
       printf("\n");
}

int inverted_triangle(int n)
{
    int i,j;
    for(i=1; i<=n; i++)
        {   
         for(j=1; j<i; j++)
         {
            printf(" ");
         }
         for(j=1; j<=(n*2 -(2*i-1)); j++)
         {
            printf("*");
         }
         printf("\n");
        }
        printf("\n");
}
int crossbox(int n)
{
    int i,j;

        for(i=1; i<=n; i++)
    {
    
        for(j=1; j<=n; j++)
        {
           
            if(i==1 || i==n || j==1 || j==n || i==j) 
            {
                printf("*");
            }
            else if(j==(n - i + 1))
            {
                printf("*");
            }
            else
            {
                printf(" ");
            }
        }
        printf("\n");
    }
}
